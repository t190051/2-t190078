\section{基本となる関数の積分}

\begin{eqnarray}
\int_{}^{}
\frac{1}{x}
dx=\log |x|+C
\\
\int_{}^{}
\sin xdx=-\cos x+C
\\
\int_{}^{}
\cos xdx=\sin x+C
\\
\int_{}^{}
\tan xdx=-\log |\cos x|+C
\\
\int_{}^{}
\log xdx=x\log x-x+C
\\
\int_{}^{}
a^xdx=
\frac{a^x}{\log a}
+C
\end{eqnarray}
